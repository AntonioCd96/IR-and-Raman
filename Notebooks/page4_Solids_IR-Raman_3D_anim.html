formulas =[]

for i in range(1,5100):
    formulas.append(data["{0}".format(i)]["data"]["chemical_formula"])
    mpid.append(data["{0}".format(i)]["key_value_pairs"]["mpid"])
    
d_look_up      = { formulas[i]:[mpid[i],f"{i+1}"] for i in range(len(formulas))}
